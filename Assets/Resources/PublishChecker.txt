set @c1 = (
    select count(*) from(
        select paid, count(author = true or null) as author
        from publish
        group by paid
    ) as tmp
    where tmp.author != 1
);

delete from paper where not exists(
	select * from publish
	where publish.paid = paper.paid
);

select if (@c1 != 0 , 1, 0);
